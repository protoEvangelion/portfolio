## Latest Posts

<SourceList filter="posts">
    {posts => (
        <Grid
            sx={{
                gridTemplateColumns: ['1fr', '1fr', '1fr 1fr'],
            }}
        >
            {posts
                .filter(edge => !edge.node.frontmatter.isPrivate)
                .reduce((posts, post) => {
                    return post.node.frontmatter.pinned ? [post, ...posts] : [...posts, post];
                }, [])
                .splice(0, 4)
                .map((edge, index) => {
                    const {
                        frontmatter: { title, tags, date, dateModified, pinned, featuredImage },
                        excerpt,
                        fields: { slug },
                    } = edge.node;
                    return (
                        <PostCard
                            key={index}
                            title={title}
                            featuredImage={featuredImage}
                            tags={tags}
                            date={date}
                            dateModified={dateModified}
                            excerpt={excerpt}
                            slug={slug}
                            pinned={pinned}
                        />
                    );
                })}
        </Grid>
    )}

</SourceList>
